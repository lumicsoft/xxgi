async function loadActiveDeposits() { const grid = document.getElementById('deposits-grid'); const capDisplay = document.getElementById('total-active-capital'); const profitDisplay = document.getElementById('total-daily-profit'); if (!window.contract || !window.signer) { setTimeout(loadActiveDeposits, 1000); return; } try { const address = await window.signer.getAddress(); // Contract ke fetchAllUserData function se user ka pura data ek sath mangwao const data = await window.contract.fetchAllUserData(address); // Stats update karein const totalInvestment = parseFloat(ethers.utils.formatUnits(data.totalSelfInvestment, 18)); capDisplay.innerText = totalInvestment.toFixed(2); // AB SABSE ZAROORI: Positions fetch karna // Contract mein Position[] ek private array ki tarah behave karti hai // Isliye hum direct loop chala kar check karenge let html = ''; let totalDailyYield = 0; let hasNodes = false; // Hum index 0 se shuru karke tab tak check karenge jab tak data mil raha hai for (let i = 0; i < 50; i++) { // Max 50 deposits maan ke chal rahe hain try { // Aapke contract mein positions access karne ka sahi syntax const p = await window.contract.users(address); // Note: Agar contract mapping positions index-wise allow nahi kar raha // toh humein positions fetch karne ke liye niche wala logic use karna hoga const pos = await window.contract.users(address).positions(i); const amount = parseFloat(ethers.utils.formatUnits(pos.amount, 18)); if (amount <= 0) break; hasNodes = true; const startTime = pos.startTime.toNumber(); const rate = pos.dailyRate.toNumber(); // e.g. 95, 111, 133 const duration = pos.duration.toNumber(); const isWithdrawn = pos.isWithdrawn; const dailyProfit = (amount * rate) / 10000; totalDailyYield += dailyProfit; const currentTime = Math.floor(Date.now() / 1000); let progress = (((currentTime - startTime) / (duration * 86400)) * 100).toFixed(2); if (progress > 100) progress = 100; if (progress < 0) progress = 0; html += `

NODE #0${i + 1}

${amount.toFixed(2)}

${isWithdrawn ? 'Completed' : 'Mining'}

Daily ROI

+${dailyProfit.toFixed(2)}

Duration

${duration} Days

PROGRESS ${progress}%
`; } catch (err) { break; // Loop end } } profitDisplay.innerText = totalDailyYield.toFixed(2); grid.innerHTML = hasNodes ? html : `
NO ACTIVE DEPOSITS FOUND
`; if (window.lucide) lucide.createIcons(); } catch (e) { console.error("Error:", e); grid.innerHTML = `
SYNC ERROR
`; } }
