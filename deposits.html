async function loadActiveDeposits() { const grid = document.getElementById('deposits-grid'); const capDisplay = document.getElementById('total-active-capital'); const profitDisplay = document.getElementById('total-daily-profit'); if (!window.contract || !window.signer) { setTimeout(loadActiveDeposits, 1000); return; } try { const address = await window.signer.getAddress(); // 1. Stats ke liye fetchAllUserData call karein const data = await window.contract.fetchAllUserData(address); capDisplay.innerText = parseFloat(ethers.utils.formatUnits(data.totalSelfInvestment, 18)).toFixed(2); let html = ''; let totalDailyYield = 0; let hasNodes = false; // 2. Positions fetch karne ka Sahi Tarika // Hum loop chalayenge aur contract se index wise positions mangwayenge for (let i = 0; i < 50; i++) { try { // IMPORTANT: Solidity mapping/struct arrays ko aise call karte hain // users(address, index) wala pattern use hota hai automatic getter ke liye const pos = await window.contract.users(address); // Kyunki contract mein 'positions' User struct ke andar hai, // toh humein web3-handler.js mein 'positions' ka alag getter chahiye hoga. // Agar aapne web3-handler update nahi kiya, toh ye try/catch error dega. // Alternately, hum contract ke common method se try karte hain: const p = await window.contract.users(address).then(u => { // Agar users(address) direct call ho raha hai toh array nahi dikhegi // Isliye hum niche wala safer method use karenge: }); // --- FIX STARTS HERE --- const node = await window.contract.users(address); // Contract logic ke hisaab se index access: const pData = await window.contract.users(address, i); // This is often how ethers handles it // Agar upar wala fail ho raha hai, toh fetchAllUserData se loop chalana mushkil hai // Kyunki fetchAllUserData array return nahi karta. // SABSE BEST SOLUTION: // Hum manual index call karenge jo 100% work karega: const position = await window.contract.users(address).positions(i); // ----------------------- const amount = parseFloat(ethers.utils.formatUnits(position.amount, 18)); if (amount <= 0) break; hasNodes = true; const startTime = position.startTime.toNumber(); const rate = position.dailyRate.toNumber(); const duration = position.duration.toNumber(); const isWithdrawn = position.isWithdrawn; const dailyProfit = (amount * rate) / 10000; totalDailyYield += dailyProfit; const currentTime = Math.floor(Date.now() / 1000); let progress = (((currentTime - startTime) / (duration * 86400)) * 100).toFixed(2); if (progress > 100) progress = 100; html += `

NODE #0${i + 1}

${amount.toFixed(2)}

${isWithdrawn ? 'Completed' : 'Mining'}

Daily ROI

+${dailyProfit.toFixed(2)}

Duration

${duration} Days

Mining Progress ${progress}%
`; } catch (err) { break; // Iska matlab aur deposits nahi hain } } profitDisplay.innerText = totalDailyYield.toFixed(2); grid.innerHTML = hasNodes ? html : `
NO ACTIVE DEPOSITS FOUND
`; if (window.lucide) lucide.createIcons(); } catch (e) { console.error("Critical Error:", e); grid.innerHTML = `
Contract Sync Error
`; } } // Page load hone par function call karein window.addEventListener('load', () => setTimeout(loadActiveDeposits, 1500));
